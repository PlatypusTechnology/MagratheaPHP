 66 function GetRelationsByObject($obj){
 67     $rels = GetRelations();
 68     $relations = array();
 69     if( count($rels) == 0 ) return $relations;
 70     $index = 0;
 71     foreach( $rels["rel_obj_base"] as $objbase ){
 72         if($objbase == $obj){
 73             array_push($relations, ExtractRelFromRelArray($rels, $index));
 74         }
 75         $index++;
 76     }
 77     return $relations;

 79 function GetRelationByName($rel_name){
