 49     public static function Run($magQuery){
 50         $array_obj = $magQuery->GetObjArray();
 51         if(count($array_obj) > 0){
 52             $objects = array();
 53             $result = static::QueryResult($magQuery->SQL());
 54             foreach ($result as $r) {
 55                 $splitResult = MagratheaQuery::SplitArrayResult($r);
 56                 $new_object = new static::$modelName();
 57                 if(count($splitResult) > 0)
 58                     $r = $splitResult[$new_object->GetDbTable()];
 59                 $new_object->LoadObjectFromTableRow($r);
 60                 foreach($array_obj as $obj){
 61                     $obj->LoadObjectFromTableRow($splitResult[$obj->GetDbTable()]);
 62                     $objname = get_class($obj);
 63                     $new_object->$objname = clone $obj;
 64                     unset($obj);
 65                 }
 66                 array_push($objects, clone $new_object);
 67             }
 68             return $objects;
