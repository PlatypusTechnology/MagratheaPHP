 21 class MagratheaView{
 22 
 23     private $javascript_files = array();
 24     private $javascript_lastmodified_arr = array();
 25     private $css_files = array();
 26     private $css_lastmodified_arr = array();
 27     private $site_path = "";
 28     private $compressionMode = "default";
 29     private $relativePath = false;
 30     private $urlForAssets = "";
 31 
 32     public function __construct(){
 33         $this->site_path = MagratheaConfig::Instance()->GetConfigFromDefault("site_path");
 34     }
 35 
 36     /*
 37     * set compression mode:
 38     * DEFAULT, ADVANCED
 39     */
 40     public function SetCompressionMode($mode){
 41         $this->compressionMode = strtolower($mode);
 42     }
 43     public function IsRelativePath($it_is){
 44         $this->relativePath = $it_is;
 45     }
 46     public function ShouldCompressJavascript(){
 47         return MagratheaConfig::Instance()->GetConfigFromDefault("compress_js");
 48     }
 49     public function ShouldCompressCss(){
 50         return MagratheaConfig::Instance()->GetConfigFromDefault("compress_css");
 51     }
 52     public function SetUrlForAssets($url){
 53         $this->urlForAssets = $url;
 54     }
 55 
 56     public function IncludeJavascript($js_file){
 57         array_push($this->javascript_files, $js_file);
 58         $file_full = $this->site_path.$js_file;
 59         if(file_exists($file_full)){
 60             $lastm = filemtime($file_full);
 61         } else {
 62             throw new MagratheaViewException("File ".$file_full." could not be reached!");
 63         }
 64         array_push($this->javascript_lastmodified_arr, $lastm);
 65     }
 66     public function IncludeCSS($css_file){
 67         array_push($this->css_files, $css_file);
 68         $file_full = $this->site_path.$css_file;
 69         if(file_exists($file_full)){
 70             $lastm = filemtime($file_full);
 71         } else {
 72             throw new MagratheaViewException("File ".$file_full." could not be reached!");
 73         }
 74         array_push($this->css_lastmodified_arr, $lastm);
 75     }

 77     public function Javascripts($compression=null){
 78         $compression = ($compression==null ? $this->ShouldCompressJavascript() : $compression );
 79         $array_files = array_unique($this->javascript_files);
 80         $jsContent = "<!--JS FILES MANAGED BY MAGRATHEA [compression: ".$compression."] -->\n";
 81         if($compression == "true"){
 82             sort($this->javascript_lastmodified_arr);
 83             $js_lmod = implode("_", $this->javascript_lastmodified_arr);
 84             $js_lmod_hash = md5($js_lmod);
 85             $compressedFileName = "javascript/_compressed/".$js_lmod_hash."_compressed.js";
 86             if(!file_exists($compressedFileName)){
 87                 if (!$handle = @fopen($compressedFileName, 'w')) { 
 88                     $jsContent .= "<!--error compressing javascript! could not create file-->";
 89                     $jsContent .= $this->Javascripts("false");
 90                     return $jsContent;
 91                 } 
 92                 $jsCompressor = new MagratheaCompressor(MagratheaCompressor::COMPRESS_JS);
 93                 $jsCompressor->setCompressionMode($this->compressionMode);
 94                 foreach($array_files as $file){
 95                     $jsCompressor->add($file);
 96                 }
 97                 $jsCompressor->compress();
 98                 $compressed_js = $jsCompressor->GetCompressedContent();
 99                 if (!fwrite($handle, $compressed_js)) { 
100                     $jsContent .= "<!--error compressing javascript! could not write file-->";
101                     $jsContent .= $this->Javascripts("false");
102                     return $jsContent;
103                 } 
104                 fclose($handle); 
105             }
106             $jsContent .= "<script type='text/javascript' src='".$this->urlForAssets.($this->relativePath ? "" : "/").$compressedFileName."'></script>\n"; 
107         } else {
108             foreach($array_files as $file){
109                 $jsContent .= "<script type='text/javascript' src='".$this->urlForAssets.($this->relativePath ? "" : "/").$file."'></script>\n"; 
110             }
111         }
112         return $jsContent;
113     }

127                     $cssContent .= $this->CSSs("false");
128                     return $cssContent;
129                 } 
130                 $cssCompressor = new MagratheaCompressor(@MagratheaCompressor::COMPRESS_CSS);
131                 foreach($array_files as $file){
132                     $cssCompressor->add($file);
133                 }
134                 $cssCompressor->compress();
135                 $compressed_css = $cssCompressor->GetCompressedContent();
136                 if (!fwrite($handle, $compressed_css)) {

139                     return $cssContent;
140                 } 
141                 fclose($handle); 
142             }
143             $cssContent .= "<link href='".$this->urlForAssets.($this->relativePath ? "" : "/").$compressedFileName."' rel='stylesheet'>\n"; 
144         } else {
145             foreach($array_files as $file){
146                 $cssContent .= "<link href='".$this->urlForAssets.($this->relativePath ? "" : "/").$file."' rel='stylesheet'>\n";
